Numeros primos en un rango de numeros

Obtener todos los numeros primos dentro de un rango de numeros ej. 2 al 5, los numeros primos entre 2 y 5 son 2,3 y 5

                        function calcNumbers(start,end){
                            let result  = [];
                            start = start < 2 ? 2 : start;
                            end   = end < 2 ? 2 : end;

                            for(let i = start; i <= end ; i++){
                                let count = 0;

                                for(let j = 2 ; j <= i ; j++){
                                    if(i % 2 === 0){
                                        count++;
                                    }

                                    if(count > 2)
                                        break;

                                }

                                if(count <= 2){
                                    result.push(i);
                                }
                            }

                            return result;
                        }
